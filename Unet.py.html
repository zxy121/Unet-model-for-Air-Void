# ==================== 增强版数据集类 ====================
class VoidDataset(Dataset):
    def __init__(self, images_dir, masks_dir, size=(IMG_SIZE, IMG_SIZE), augment=False, augment_factor=1):
        self.images_dir = images_dir
        self.masks_dir = masks_dir
        self.size = size
        self.augment = augment
        self.augment_factor = augment_factor  # 增强倍数
        self.images = sorted([f for f in os.listdir(images_dir)
                              if f.endswith(('.png', '.jpg', '.jpeg'))])

    def __len__(self):
        if self.augment:
            return len(self.images) * self.augment_factor
        return len(self.images)

    def __getitem__(self, idx):
        if self.augment:
            # 增强模式：每张原图生成 augment_factor 个变体
            original_idx = idx % len(self.images)
            aug_idx = idx // len(self.images)
        else:
            original_idx = idx
            aug_idx = 0

        img_name = self.images[original_idx]
        base_name = os.path.splitext(img_name)[0]

        img_path = os.path.join(self.images_dir, img_name)
        image = cv2.imread(img_path)
        image = cv2.cvtColor(image, cv2.COLOR_BGR2RGB)

        mask_path = os.path.join(self.masks_dir, base_name + '.png')
        mask = cv2.imread(mask_path, cv2.IMREAD_GRAYSCALE)

        # 数据增强（根据aug_idx不同，生成不同的增强版本）
        if self.augment:
            # 使用aug_idx作为随机种子，确保每个epoch的增强版本不同但可复现
            np.random.seed(aug_idx * 1000 + original_idx)

            # 1. 随机翻转
            if np.random.random() > 0.5:
                image = np.fliplr(image).copy()
                mask = np.fliplr(mask).copy()
            if np.random.random() > 0.5:
                image = np.flipud(image).copy()
                mask = np.flipud(mask).copy()

            # 2. 随机旋转
            if np.random.random() > 0.5:
                angle = np.random.randint(-15, 15)
                h, w = image.shape[:2]
                center = (w // 2, h // 2)
                M = cv2.getRotationMatrix2D(center, angle, 1.0)
                image = cv2.warpAffine(image, M, (w, h), borderMode=cv2.BORDER_REFLECT)
                mask = cv2.warpAffine(mask, M, (w, h), borderMode=cv2.BORDER_REFLECT)

            # 3. 随机缩放
            if np.random.random() > 0.5:
                scale = np.random.uniform(0.8, 1.2)
                new_w, new_h = int(image.shape[1] * scale), int(image.shape[0] * scale)
                image = cv2.resize(image, (new_w, new_h))
                mask = cv2.resize(mask, (new_w, new_h), interpolation=cv2.INTER_NEAREST)

            # 4. 亮度调整
            if np.random.random() > 0.5:
                brightness = np.random.uniform(0.6, 1.4)
                image = np.clip(image * brightness, 0, 255).astype(np.uint8)

            # 5. 对比度调整
            if np.random.random() > 0.5:
                contrast = np.random.uniform(0.7, 1.3)
                mean = np.mean(image, axis=(0, 1), keepdims=True)
                image = np.clip((image - mean) * contrast + mean, 0, 255).astype(np.uint8)

            # 6. 高斯噪声
            if np.random.random() > 0.7:
                noise = np.random.normal(0, 10, image.shape)
                image = np.clip(image + noise, 0, 255).astype(np.uint8)

        # 缩放到目标尺寸
        image = cv2.resize(image, self.size)
        mask = cv2.resize(mask, self.size, interpolation=cv2.INTER_NEAREST)
        mask = (mask > 127).astype(np.float32)

        # 归一化
        image = image.astype(np.float32) / 255.0
        image = image.transpose(2, 0, 1)
        mask = np.expand_dims(mask, axis=0)

        return torch.from_numpy(image), torch.from_numpy(mask)


# ==================== Combo Loss ====================
class ComboLoss(nn.Module):
    def __init__(self, pos_weight=20.0):
        super().__init__()
        self.pos_weight = pos_weight

    def dice_loss(self, pred, target, smooth=1e-5):
        pred = torch.sigmoid(pred)
        pred = pred.contiguous().view(-1)
        target = target.contiguous().view(-1)
        intersection = (pred * target).sum()
        dice = (2. * intersection + smooth) / (pred.sum() + target.sum() + smooth)
        return 1 - dice

    def weighted_bce(self, pred, target):
        weight = torch.where(target == 1, self.pos_weight, 1.0)
        return F.binary_cross_entropy_with_logits(pred, target, weight=weight)

    def forward(self, pred, target):
        return self.dice_loss(pred, target) + self.weighted_bce(pred, target)


# ==================== U-Net模型 ====================
class DoubleConv(nn.Module):
    def __init__(self, in_channels, out_channels):
        super().__init__()
        self.double_conv = nn.Sequential(
            nn.Conv2d(in_channels, out_channels, kernel_size=3, padding=1),
            nn.BatchNorm2d(out_channels),
            nn.ReLU(inplace=True),
            nn.Conv2d(out_channels, out_channels, kernel_size=3, padding=1),
            nn.BatchNorm2d(out_channels),
            nn.ReLU(inplace=True)
        )

    def forward(self, x):
        return self.double_conv(x)


class Down(nn.Module):
    def __init__(self, in_channels, out_channels):
        super().__init__()
        self.maxpool_conv = nn.Sequential(
            nn.MaxPool2d(2),
            DoubleConv(in_channels, out_channels)
        )

    def forward(self, x):
        return self.maxpool_conv(x)


class Up(nn.Module):
    def __init__(self, in_channels, out_channels, bilinear=True):
        super().__init__()
        if bilinear:
            self.up = nn.Upsample(scale_factor=2, mode='bilinear', align_corners=True)
        else:
            self.up = nn.ConvTranspose2d(in_channels, in_channels // 2, kernel_size=2, stride=2)
        self.conv = DoubleConv(in_channels, out_channels)

    def forward(self, x1, x2):
        x1 = self.up(x1)
        diffY = x2.size()[2] - x1.size()[2]
        diffX = x2.size()[3] - x1.size()[3]
        x1 = F.pad(x1, [diffX // 2, diffX - diffX // 2, diffY // 2, diffY - diffY // 2])
        x = torch.cat([x2, x1], dim=1)
        return self.conv(x)


class UNet(nn.Module):
    def __init__(self, n_channels=3, n_classes=1, bilinear=True):
        super().__init__()
        self.inc = DoubleConv(n_channels, 64)
        self.down1 = Down(64, 128)
        self.down2 = Down(128, 256)
        self.down3 = Down(256, 512)
        factor = 2 if bilinear else 1
        self.down4 = Down(512, 1024 // factor)
        self.up1 = Up(1024, 512 // factor, bilinear)
        self.up2 = Up(512, 256 // factor, bilinear)
        self.up3 = Up(256, 128 // factor, bilinear)
        self.up4 = Up(128, 64, bilinear)
        self.outc = nn.Conv2d(64, n_classes, kernel_size=1)

    def forward(self, x):
        x1 = self.inc(x)
        x2 = self.down1(x1)
        x3 = self.down2(x2)
        x4 = self.down3(x3)
        x5 = self.down4(x4)
        x = self.up1(x5, x4)
        x = self.up2(x, x3)
        x = self.up3(x, x2)
        x = self.up4(x, x1)
        return self.outc(x)


# ==================== 科学置信度计算 ====================
def compute_confidence(pred_probs, mask_contours, min_area=100):
    """ 
    置信度 = 该连通域内所有像素预测概率的平均值 
    """
    h, w = pred_probs.shape
    detections = []

    for contour in mask_contours:
        area = cv2.contourArea(contour)
        if area < min_area:
            continue

        # 创建该连通域的掩膜
        contour_mask = np.zeros((h, w), dtype=np.uint8)
        cv2.drawContours(contour_mask, [contour], -1, 1, thickness=cv2.FILLED)

        # 提取该区域内的预测概率
        region_probs = pred_probs[contour_mask == 1]

        if len(region_probs) > 0:
            confidence = np.mean(region_probs)
            x, y, w_box, h_box = cv2.boundingRect(contour)

            detections.append({
                'bbox': [int(x), int(y), int(w_box), int(h_box)],
                'confidence': float(confidence),
                'area': int(area)
            })

    detections.sort(key=lambda x: x['confidence'], reverse=True)
    return detections


# ==================== 评估函数 ====================
def calculate_metrics(pred_logits, true_mask, threshold=0.3):
    pred_prob = torch.sigmoid(pred_logits)
    pred_mask = (pred_prob > threshold).float()

    pred_flat = pred_mask.cpu().numpy().flatten()
    true_flat = true_mask.cpu().numpy().flatten()

    tn, fp, fn, tp = confusion_matrix(true_flat, pred_flat, labels=[0, 1]).ravel()

    precision = tp / (tp + fp) if (tp + fp) > 0 else 0
    recall = tp / (tp + fn) if (tp + fn) > 0 else 0
    f1 = 2 * precision * recall / (precision + recall) if (precision + recall) > 0 else 0
    iou = tp / (tp + fp + fn) if (tp + fp + fn) > 0 else 0

    return {
        'precision': precision,
        'recall': recall,
        'f1': f1,
        'iou': iou,
        'tp': tp, 'fp': fp, 'fn': fn, 'tn': tn
    }


# ==================== 训练函数 ====================
def train():
    print("\n加载数据集...")

    # 计算正样本权重
    sample_mask = cv2.imread(os.path.join(MASKS_FOLDER, os.listdir(MASKS_FOLDER)[0]), cv2.IMREAD_GRAYSCALE)
    void_ratio = np.mean(sample_mask > 0)
    pos_weight = max((1 - void_ratio) / void_ratio, 10.0)
    print(f"空隙像素占比: {void_ratio:.4f}, 正样本权重: {pos_weight:.2f}")

    # 数据增强倍数（每张图生成5个变体，52*5=260张/epoch）
    AUGMENT_FACTOR = 5

    # 创建数据集
    train_dataset = VoidDataset(IMAGES_FOLDER, MASKS_FOLDER, size=(IMG_SIZE, IMG_SIZE),
                                augment=True, augment_factor=AUGMENT_FACTOR)
    val_dataset = VoidDataset(IMAGES_FOLDER, MASKS_FOLDER, size=(IMG_SIZE, IMG_SIZE),
                              augment=False, augment_factor=1)

    # 划分训练集和验证集（基于原始图片索引）
    from torch.utils.data import Subset
    all_indices = list(range(len(val_dataset)))  # val_dataset是原始图片数
    np.random.seed(42)
    np.random.shuffle(all_indices)
    split_idx = int(len(all_indices) * 0.8)
    train_original_indices = all_indices[:split_idx]
    val_indices = all_indices[split_idx:]

    # 构建训练集的增强版本索引
    train_augmented_indices = []
    for orig_idx in train_original_indices:
        for aug_idx in range(AUGMENT_FACTOR):
            train_augmented_indices.append(orig_idx * AUGMENT_FACTOR + aug_idx)

    train_loader = DataLoader(train_dataset, batch_size=BATCH_SIZE,
                              sampler=torch.utils.data.SubsetRandomSampler(train_augmented_indices),
                              num_workers=4, pin_memory=True)
    val_loader = DataLoader(Subset(val_dataset, val_indices), batch_size=BATCH_SIZE,
                            shuffle=False, num_workers=4, pin_memory=True)

    print(f"原始训练集: {len(train_original_indices)} 张")
    print(f"增强后训练集: {len(train_augmented_indices)} 张/epoch (增强倍数={AUGMENT_FACTOR})")
    print(f"验证集: {len(val_indices)} 张")

    # 模型
    model = UNet().to(DEVICE)
    optimizer = torch.optim.AdamW(model.parameters(), lr=LEARNING_RATE, weight_decay=1e-4)
    scheduler = torch.optim.lr_scheduler.CosineAnnealingLR(optimizer, T_max=EPOCHS)
    criterion = ComboLoss(pos_weight=pos_weight)

    print(f"\n模型参数量: {sum(p.numel() for p in model.parameters()):,}")
    print(f"\n开始训练 {EPOCHS} 个epoch...")
    print("=" * 60)

    best_f1 = 0
    best_metrics = None
    train_losses = []
    val_losses = []
    f1_scores = []

    start_time = time.time()

    for epoch in range(EPOCHS):
        # 训练
        model.train()
        train_loss = 0
        for images, masks in train_loader:
            images, masks = images.to(DEVICE), masks.to(DEVICE)
            optimizer.zero_grad()
            outputs = model(images)
            loss = criterion(outputs, masks)
            loss.backward()
            optimizer.step()
            train_loss += loss.item()

        avg_train_loss = train_loss / len(train_loader)
        train_losses.append(avg_train_loss)

        # 验证
        model.eval()
        val_loss = 0
        all_outputs = []
        all_targets = []

        with torch.no_grad():
            for images, masks in val_loader:
                images, masks = images.to(DEVICE), masks.to(DEVICE)
                outputs = model(images)
                loss = criterion(outputs, masks)
                val_loss += loss.item()
                all_outputs.append(outputs)
                all_targets.append(masks)

        avg_val_loss = val_loss / len(val_loader)
        val_losses.append(avg_val_loss)

        # 合并所有验证结果
        all_outputs = torch.cat(all_outputs, dim=0)
        all_targets = torch.cat(all_targets, dim=0)

        # 计算指标
        metrics = calculate_metrics(all_outputs, all_targets, threshold=CONFIDENCE_THRESHOLD)
        f1_scores.append(metrics['f1'])

        # 保存最佳模型
        if metrics['f1'] > best_f1:
            best_f1 = metrics['f1']
            best_metrics = metrics
            torch.save(model.state_dict(), os.path.join(OUTPUT_FOLDER, 'best_model.pth'))

        scheduler.step()

        if (epoch + 1) % 10 == 0:
            print(
                f"Epoch {epoch + 1:3d}/{EPOCHS} | Train Loss: {avg_train_loss:.4f} | Val Loss: {avg_val_loss:.4f} | F1: {metrics['f1']:.4f} | Prec: {metrics['precision']:.4f} | Rec: {metrics['recall']:.4f}")

    training_time = time.time() - start_time

    print("\n" + "=" * 60)
    print(f"训练完成！")
    print(f"最佳F1分数: {best_f1:.4f}")
    print(f"对应精确率: {best_metrics['precision']:.4f}")
    print(f"对应召回率: {best_metrics['recall']:.4f}")
    print(f"对应IoU: {best_metrics['iou']:.4f}")
    print(f"训练时间: {training_time / 60:.2f}分钟")
    print("=" * 60)

    # 绘制曲线
    plt.figure(figsize=(12, 5))
    plt.subplot(1, 2, 1)
    plt.plot(train_losses, label='Train Loss')
    plt.plot(val_losses, label='Val Loss')
    plt.xlabel('Epoch')
    plt.ylabel('Loss')
    plt.title('Loss Curves')
    plt.legend()
    plt.grid(True)

    plt.subplot(1, 2, 2)
    plt.plot(f1_scores, label='F1 Score')
    plt.xlabel('Epoch')
    plt.ylabel('F1')
    plt.title(f'F1 Score Curve (Best: {best_f1:.4f})')
    plt.legend()
    plt.grid(True)

    plt.savefig(os.path.join(OUTPUT_FOLDER, 'curves.png'), dpi=150)
    plt.close()

    # 生成置信度可视化
    print("\n生成置信度可视化...")
    generate_confidence_visualization(model, val_loader)

    print(f"\n所有结果保存至: {OUTPUT_FOLDER}")


# ==================== 置信度可视化 ====================
def generate_confidence_visualization(model, val_loader):
    """生成带置信度标注的检测结果图"""
    model.eval()

    detection_folder = os.path.join(OUTPUT_FOLDER, 'detections')
    os.makedirs(detection_folder, exist_ok=True)

    with torch.no_grad():
        for batch_idx, (images, masks) in enumerate(val_loader):
            if batch_idx >= 4:  # 只生成前4张
                break

            images = images.to(DEVICE)
            outputs = model(images)
            probs = torch.sigmoid(outputs).cpu().numpy()

            for i in range(images.shape[0]):
                prob = probs[i, 0]
                pred = (prob > CONFIDENCE_THRESHOLD).astype(np.uint8)

                # 提取轮廓
                contours, _ = cv2.findContours(pred, cv2.RETR_EXTERNAL, cv2.CHAIN_APPROX_SIMPLE)

                # 计算置信度
                detections = compute_confidence(prob, contours, min_area=100)

                # 显示原图（需要反归一化）
                img_display = images[i].cpu().numpy().transpose(1, 2, 0)

                # 绘制检测框
                img_with_boxes = (img_display * 255).astype(np.uint8).copy()
                img_with_boxes = cv2.cvtColor(img_with_boxes, cv2.COLOR_RGB2BGR)

                for det in detections:
                    x, y, w, h = det['bbox']
                    conf = det['confidence']

                    if conf >= 0.7:
                        color = (0, 255, 0)
                    elif conf >= 0.5:
                        color = (0, 255, 255)
                    else:
                        color = (0, 165, 255)

                    cv2.rectangle(img_with_boxes, (x, y), (x + w, y + h), color, 2)
                    label = f"void {conf:.3f}"
                    cv2.putText(img_with_boxes, label, (x, y - 5),
                                cv2.FONT_HERSHEY_SIMPLEX, 0.5, color, 1)

                output_path = os.path.join(detection_folder, f'detection_{batch_idx}_{i}.png')
                cv2.imwrite(output_path, img_with_boxes)

    print(f"  已保存检测图到 {detection_folder}/")


# ==================== 主函数 ====================
if __name__ == "__main__":
    train()
