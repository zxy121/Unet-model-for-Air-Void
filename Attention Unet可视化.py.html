# ==================== 数据集类 ====================
class VoidDataset(Dataset):
    def __init__(self, images_dir, masks_dir, size=(IMG_SIZE, IMG_SIZE)):
        self.images_dir = images_dir
        self.masks_dir = masks_dir
        self.size = size
        self.images = sorted([f for f in os.listdir(images_dir)
                              if f.endswith(('.png', '.jpg', '.jpeg'))])

    def __len__(self):
        return len(self.images)

    def __getitem__(self, idx):
        img_name = self.images[idx]
        base_name = os.path.splitext(img_name)[0]
        img_path = os.path.join(self.images_dir, img_name)
        image = cv2.imread(img_path)
        image = cv2.cvtColor(image, cv2.COLOR_BGR2RGB)
        original_h, original_w = image.shape[:2]
        image = cv2.resize(image, self.size)
        mask_path = os.path.join(self.masks_dir, base_name + '.png')
        mask = cv2.imread(mask_path, cv2.IMREAD_GRAYSCALE)
        mask = cv2.resize(mask, self.size, interpolation=cv2.INTER_NEAREST)
        mask = (mask > 127).astype(np.float32)
        image = image.astype(np.float32) / 255.0
        image = image.transpose(2, 0, 1)
        mask = np.expand_dims(mask, axis=0)
        return torch.from_numpy(image), torch.from_numpy(mask), img_name, original_h, original_w


# ==================== Attention U-Net 模型 ====================
class DoubleConv(nn.Module):
    def __init__(self, in_channels, out_channels):
        super().__init__()
        self.double_conv = nn.Sequential(
            nn.Conv2d(in_channels, out_channels, kernel_size=3, padding=1),
            nn.BatchNorm2d(out_channels),
            nn.ReLU(inplace=True),
            nn.Conv2d(out_channels, out_channels, kernel_size=3, padding=1),
            nn.BatchNorm2d(out_channels),
            nn.ReLU(inplace=True)
        )

    def forward(self, x):
        return self.double_conv(x)


class Down(nn.Module):
    def __init__(self, in_channels, out_channels):
        super().__init__()
        self.maxpool_conv = nn.Sequential(
            nn.MaxPool2d(2),
            DoubleConv(in_channels, out_channels)
        )

    def forward(self, x):
        return self.maxpool_conv(x)


class Up(nn.Module):
    def __init__(self, in_channels, out_channels):
        super().__init__()
        self.up = nn.ConvTranspose2d(in_channels, in_channels // 2, kernel_size=2, stride=2)
        self.conv = DoubleConv(in_channels, out_channels)

    def forward(self, x1, x2):
        x1 = self.up(x1)
        diffY = x2.size()[2] - x1.size()[2]
        diffX = x2.size()[3] - x1.size()[3]
        x1 = F.pad(x1, [diffX // 2, diffX - diffX // 2, diffY // 2, diffY - diffY // 2])
        x = torch.cat([x2, x1], dim=1)
        return self.conv(x)


class AttentionBlock(nn.Module):
    def __init__(self, F_g, F_l, F_int):
        super(AttentionBlock, self).__init__()
        self.W_g = nn.Sequential(
            nn.Conv2d(F_g, F_int, kernel_size=1, stride=1, padding=0, bias=True),
            nn.BatchNorm2d(F_int)
        )
        self.W_x = nn.Sequential(
            nn.Conv2d(F_l, F_int, kernel_size=1, stride=1, padding=0, bias=True),
            nn.BatchNorm2d(F_int)
        )
        self.psi = nn.Sequential(
            nn.Conv2d(F_int, 1, kernel_size=1, stride=1, padding=0, bias=True),
            nn.BatchNorm2d(1),
            nn.Sigmoid()
        )
        self.relu = nn.ReLU(inplace=True)

    def forward(self, g, x):
        g1 = self.W_g(g)
        x1 = self.W_x(x)
        psi = self.relu(g1 + x1)
        psi = self.psi(psi)
        return x * psi


class AttentionUNet(nn.Module):
    def __init__(self, n_channels=3, n_classes=1):
        super(AttentionUNet, self).__init__()

        self.conv1 = DoubleConv(n_channels, 64)
        self.pool1 = nn.MaxPool2d(2)
        self.conv2 = DoubleConv(64, 128)
        self.pool2 = nn.MaxPool2d(2)
        self.conv3 = DoubleConv(128, 256)
        self.pool3 = nn.MaxPool2d(2)
        self.conv4 = DoubleConv(256, 512)
        self.pool4 = nn.MaxPool2d(2)
        self.conv5 = DoubleConv(512, 1024)

        self.att4 = AttentionBlock(F_g=512, F_l=512, F_int=256)
        self.att3 = AttentionBlock(F_g=256, F_l=256, F_int=128)
        self.att2 = AttentionBlock(F_g=128, F_l=128, F_int=64)
        self.att1 = AttentionBlock(F_g=64, F_l=64, F_int=32)

        self.up4 = Up(1024, 512)
        self.up3 = Up(512, 256)
        self.up2 = Up(256, 128)
        self.up1 = Up(128, 64)

        self.final = nn.Conv2d(64, n_classes, kernel_size=1)

    def forward(self, x):
        x1 = self.conv1(x)
        x2 = self.conv2(self.pool1(x1))
        x3 = self.conv3(self.pool2(x2))
        x4 = self.conv4(self.pool3(x3))
        x5 = self.conv5(self.pool4(x4))

        d4 = self.up4(x5, x4)
        d4 = self.att4(g=d4, x=x4)
        d3 = self.up3(d4, x3)
        d3 = self.att3(g=d3, x=x3)
        d2 = self.up2(d3, x2)
        d2 = self.att2(g=d2, x=x2)
        d1 = self.up1(d2, x1)
        d1 = self.att1(g=d1, x=x1)

        out = self.final(d1)
        return out


# ==================== 评估函数 ====================
def calculate_metrics(pred_logits, true_mask, threshold=0.3):
    pred_prob = torch.sigmoid(pred_logits)
    pred_mask = (pred_prob > threshold).float()
    pred_flat = pred_mask.cpu().numpy().flatten()
    true_flat = true_mask.cpu().numpy().flatten()
    tn, fp, fn, tp = confusion_matrix(true_flat, pred_flat, labels=[0, 1]).ravel()
    precision = tp / (tp + fp) if (tp + fp) > 0 else 0
    recall = tp / (tp + fn) if (tp + fn) > 0 else 0
    f1 = 2 * precision * recall / (precision + recall) if (precision + recall) > 0 else 0
    iou = tp / (tp + fp + fn) if (tp + fp + fn) > 0 else 0
    return {'precision': precision, 'recall': recall, 'f1': f1, 'iou': iou,
            'tp': tp, 'fp': fp, 'fn': fn, 'tn': tn}


# ==================== 科学置信度计算 ====================
def compute_confidence(pred_probs, mask_contours, min_area=100):
    h, w = pred_probs.shape
    detections = []
    for contour in mask_contours:
        area = cv2.contourArea(contour)
        if area < min_area:
            continue
        contour_mask = np.zeros((h, w), dtype=np.uint8)
        cv2.drawContours(contour_mask, [contour], -1, 1, thickness=cv2.FILLED)
        region_probs = pred_probs[contour_mask == 1]
        if len(region_probs) > 0:
            confidence = np.mean(region_probs)
            x, y, w_box, h_box = cv2.boundingRect(contour)
            detections.append({
                'bbox': [int(x), int(y), int(w_box), int(h_box)],
                'confidence': float(confidence),
                'area': int(area)
            })
    detections.sort(key=lambda x: x['confidence'], reverse=True)
    return detections


# ==================== 主函数 ====================
def main():
    print("\n1. 加载模型...")
    model = AttentionUNet(n_channels=3, n_classes=1).to(DEVICE)
    model.load_state_dict(torch.load(MODEL_PATH, map_location=DEVICE))
    model.eval()
    print("   模型加载成功！")

    print("\n2. 加载验证集...")
    full_dataset = VoidDataset(IMAGES_FOLDER, MASKS_FOLDER)
    all_indices = list(range(len(full_dataset)))
    np.random.seed(42)
    np.random.shuffle(all_indices)
    split_idx = int(len(all_indices) * 0.8)
    val_indices = all_indices[split_idx:]
    val_dataset = Subset(full_dataset, val_indices)
    val_loader = DataLoader(val_dataset, batch_size=BATCH_SIZE, shuffle=False)
    print(f"   验证集大小: {len(val_dataset)} 张")

    # 收集预测结果
    print("\n3. 收集预测结果...")
    all_outputs = []
    all_targets = []
    all_probs = []

    with torch.no_grad():
        for images, masks, names, h, w in val_loader:
            images = images.to(DEVICE)
            outputs = model(images)
            probs = torch.sigmoid(outputs)
            all_outputs.append(outputs.cpu())
            all_targets.append(masks)
            all_probs.append(probs.cpu())

    all_outputs = torch.cat(all_outputs, dim=0)
    all_targets = torch.cat(all_targets, dim=0)
    all_probs = torch.cat(all_probs, dim=0)

    print("\n   验证集评估结果:")
    for th in [0.3, 0.4, 0.5]:
        m = calculate_metrics(all_outputs, all_targets, threshold=th)
        print(
            f"     阈值={th}: F1={m['f1']:.4f}, Precision={m['precision']:.4f}, Recall={m['recall']:.4f}, IoU={m['iou']:.4f}")

    # 1. 混淆矩阵
    print("\n4. 生成混淆矩阵...")
    fig, axes = plt.subplots(1, 3, figsize=(15, 5))
    for i, th in enumerate([0.3, 0.4, 0.5]):
        m = calculate_metrics(all_outputs, all_targets, threshold=th)
        cm = np.array([[m['tn'], m['fp']], [m['fn'], m['tp']]])
        axes[i].imshow(cm, interpolation='nearest', cmap=plt.cm.Blues)
        axes[i].set_title(f'Attention U-Net th={th}\nF1={m["f1"]:.4f}, IoU={m["iou"]:.4f}')
        axes[i].set_xticks([0, 1])
        axes[i].set_yticks([0, 1])
        axes[i].set_xticklabels(['Background', 'Void'])
        axes[i].set_yticklabels(['Background', 'Void'])
        for j in range(2):
            for k in range(2):
                axes[i].text(k, j, format(cm[j, k], ','), ha="center", va="center")
        axes[i].set_ylabel('True Label')
        axes[i].set_xlabel('Predicted Label')
    plt.tight_layout()
    plt.savefig(os.path.join(OUTPUT_FOLDER, '1_confusion_matrices.png'), dpi=150)
    plt.close()
    print("   已保存: 1_confusion_matrices.png")

    # 2. PR曲线
    print("\n5. 生成PR曲线...")
    all_probs_flat = all_probs.numpy().flatten()
    all_targets_flat = all_targets.numpy().flatten()
    precision, recall, _ = precision_recall_curve(all_targets_flat, all_probs_flat)
    ap = average_precision_score(all_targets_flat, all_probs_flat)

    plt.figure(figsize=(8, 6))
    plt.plot(recall, precision, 'b-', linewidth=2)
    plt.fill_between(recall, precision, alpha=0.3)
    plt.xlabel('Recall')
    plt.ylabel('Precision')
    plt.title(f'Attention U-Net PR Curve (AP={ap:.4f})')
    plt.grid(True)
    plt.savefig(os.path.join(OUTPUT_FOLDER, '2_pr_curve.png'), dpi=150)
    plt.close()
    print(f"   已保存: 2_pr_curve.png (AP={ap:.4f})")

    # 3. 阈值对比图
    print("\n6. 生成阈值对比图...")
    th_values = np.arange(0.1, 0.9, 0.05)
    f1_vals, prec_vals, rec_vals = [], [], []
    for th in th_values:
        m = calculate_metrics(all_outputs, all_targets, threshold=th)
        f1_vals.append(m['f1'])
        prec_vals.append(m['precision'])
        rec_vals.append(m['recall'])

    plt.figure(figsize=(10, 6))
    plt.plot(th_values, f1_vals, 'r-', linewidth=2, label='F1 Score')
    plt.plot(th_values, prec_vals, 'g-', linewidth=2, label='Precision')
    plt.plot(th_values, rec_vals, 'b-', linewidth=2, label='Recall')
    plt.xlabel('Confidence Threshold')
    plt.ylabel('Score')
    plt.title('Attention U-Net Performance vs Threshold')
    plt.legend()
    plt.grid(True)
    best_idx = np.argmax(f1_vals)
    plt.axvline(x=th_values[best_idx], color='red', linestyle='--',
                label=f'Best F1={f1_vals[best_idx]:.4f} @ th={th_values[best_idx]:.2f}')
    plt.legend()
    plt.savefig(os.path.join(OUTPUT_FOLDER, '3_threshold_comparison.png'), dpi=150)
    plt.close()
    print("   已保存: 3_threshold_comparison.png")

    # 4. 预测对比图
    print("\n7. 生成预测对比图...")
    indices = np.random.choice(len(val_dataset), min(4, len(val_dataset)), replace=False)

    fig, axes = plt.subplots(len(indices), 4, figsize=(16, 4 * len(indices)))
    for i, idx in enumerate(indices):
        img, mask, name, h, w = val_dataset[idx]
        img_tensor = img.unsqueeze(0).to(DEVICE)
        with torch.no_grad():
            output = model(img_tensor)
            prob = torch.sigmoid(output).squeeze().cpu().numpy()
            pred = (prob > 0.3).astype(np.uint8)

        img_display = img.numpy().transpose(1, 2, 0)
        mask_display = mask.squeeze().numpy()

        axes[i, 0].imshow(img_display)
        axes[i, 0].set_title(f'Original: {name[:20]}')
        axes[i, 0].axis('off')

        axes[i, 1].imshow(mask_display, cmap='gray')
        axes[i, 1].set_title('Ground Truth')
        axes[i, 1].axis('off')

        axes[i, 2].imshow(pred, cmap='gray')
        axes[i, 2].set_title('Prediction')
        axes[i, 2].axis('off')

        axes[i, 3].imshow(prob, cmap='hot', vmin=0, vmax=1)
        axes[i, 3].set_title('Probability Heatmap')
        axes[i, 3].axis('off')

    plt.tight_layout()
    plt.savefig(os.path.join(OUTPUT_FOLDER, '4_predictions.png'), dpi=150)
    plt.close()
    print("   已保存: 4_predictions.png")

    # 5. 检测结果图
    print("\n8. 生成检测结果图...")
    det_folder = os.path.join(OUTPUT_FOLDER, 'detections')
    os.makedirs(det_folder, exist_ok=True)

    for idx in range(min(10, len(val_dataset))):
        img, mask, name, h, w = val_dataset[idx]
        img_tensor = img.unsqueeze(0).to(DEVICE)
        with torch.no_grad():
            prob = torch.sigmoid(model(img_tensor)).squeeze().cpu().numpy()

        prob_resized = cv2.resize(prob, (w, h))
        pred_resized = (prob_resized > 0.3).astype(np.uint8)

        original_img = cv2.imread(os.path.join(IMAGES_FOLDER, name))
        original_img = cv2.cvtColor(original_img, cv2.COLOR_BGR2RGB)

        contours, _ = cv2.findContours(pred_resized, cv2.RETR_EXTERNAL, cv2.CHAIN_APPROX_SIMPLE)
        detections = compute_confidence(prob_resized, contours, min_area=100)

        img_with_boxes = original_img.copy()
        for det in detections:
            conf = det['confidence']
            x, y, bw, bh = det['bbox']
            color = (0, 255, 0) if conf >= 0.7 else ((0, 255, 255) if conf >= 0.5 else (0, 165, 255))
            cv2.rectangle(img_with_boxes, (x, y), (x + bw, y + bh), color, 2)
            cv2.putText(img_with_boxes, f"void {conf:.3f}", (x, y - 5),
                        cv2.FONT_HERSHEY_SIMPLEX, 0.5, color, 1)

        cv2.imwrite(os.path.join(det_folder, f'detection_{name}'), cv2.cvtColor(img_with_boxes, cv2.COLOR_RGB2BGR))

    print(f"   已保存 {len(os.listdir(det_folder))} 张检测图")

    print("\n" + "=" * 60)
    print(f"可视化完成！结果保存至: {OUTPUT_FOLDER}")
    print("=" * 60)


if __name__ == "__main__":
    main()
