import torch
import torch.nn as nn
import torch.nn.functional as F
from torch.utils.data import Dataset, DataLoader, Subset
import cv2
import numpy as np
import os
import matplotlib.pyplot as plt
from sklearn.metrics import confusion_matrix
import time
import warnings

warnings.filterwarnings('ignore')

print("=" * 70)
print("芯样切片图像空隙语义分割 - Attention U-Net 训练")

# ==================== 配置参数 ====================
PROJECT_ROOT = '/root/autodl-tmp'
DATA_FOLDER = os.path.join(PROJECT_ROOT, 'data')
IMAGES_FOLDER = os.path.join(DATA_FOLDER, 'images')
MASKS_FOLDER = os.path.join(DATA_FOLDER, 'masks')
OUTPUT_FOLDER = os.path.join(PROJECT_ROOT, 'results_attention')
os.makedirs(OUTPUT_FOLDER, exist_ok=True)

IMG_SIZE = 256
BATCH_SIZE = 8
EPOCHS = 100
LEARNING_RATE = 0.001
CONFIDENCE_THRESHOLD = 0.3
DEVICE = torch.device('cuda' if torch.cuda.is_available() else 'cpu')

print(f"\n配置:")
print(f"  图像尺寸: {IMG_SIZE}×{IMG_SIZE}")
print(f"  批大小: {BATCH_SIZE}")
print(f"  训练轮数: {EPOCHS}")
print(f"  设备: {DEVICE}")


# ==================== 数据集类 ====================
class VoidDataset(Dataset):
    def __init__(self, images_dir, masks_dir, size=(IMG_SIZE, IMG_SIZE), augment=False, augment_factor=5):
        self.images_dir = images_dir
        self.masks_dir = masks_dir
        self.size = size
        self.augment = augment
        self.augment_factor = augment_factor
        self.images = sorted([f for f in os.listdir(images_dir)
                              if f.endswith(('.png', '.jpg', '.jpeg'))])

    def __len__(self):
        if self.augment:
            return len(self.images) * self.augment_factor
        return len(self.images)

    def __getitem__(self, idx):
        if self.augment:
            original_idx = idx % len(self.images)
            aug_idx = idx // len(self.images)
        else:
            original_idx = idx
            aug_idx = 0

        img_name = self.images[original_idx]
        base_name = os.path.splitext(img_name)[0]
        img_path = os.path.join(self.images_dir, img_name)
        image = cv2.imread(img_path)
        image = cv2.cvtColor(image, cv2.COLOR_BGR2RGB)
        mask_path = os.path.join(self.masks_dir, base_name + '.png')
        mask = cv2.imread(mask_path, cv2.IMREAD_GRAYSCALE)

        if self.augment:
            np.random.seed(aug_idx * 1000 + original_idx)

            if np.random.random() > 0.5:
                image = np.fliplr(image).copy()
                mask = np.fliplr(mask).copy()
            if np.random.random() > 0.5:
                image = np.flipud(image).copy()
                mask = np.flipud(mask).copy()
            if np.random.random() > 0.5:
                k = np.random.choice([1, 2, 3])
                image = np.rot90(image, k).copy()
                mask = np.rot90(mask, k).copy()
            if np.random.random() > 0.5:
                brightness = np.random.uniform(0.7, 1.3)
                image = np.clip(image * brightness, 0, 255).astype(np.uint8)

        image = cv2.resize(image, self.size)
        mask = cv2.resize(mask, self.size, interpolation=cv2.INTER_NEAREST)
        mask = (mask > 127).astype(np.float32)
        image = image.astype(np.float32) / 255.0
        image = image.transpose(2, 0, 1)
        mask = np.expand_dims(mask, axis=0)
        return torch.from_numpy(image), torch.from_numpy(mask)


# ==================== 基础模块 ====================
class DoubleConv(nn.Module):
    def __init__(self, in_channels, out_channels):
        super().__init__()
        self.double_conv = nn.Sequential(
            nn.Conv2d(in_channels, out_channels, kernel_size=3, padding=1),
            nn.BatchNorm2d(out_channels),
            nn.ReLU(inplace=True),
            nn.Conv2d(out_channels, out_channels, kernel_size=3, padding=1),
            nn.BatchNorm2d(out_channels),
            nn.ReLU(inplace=True)
        )

    def forward(self, x):
        return self.double_conv(x)


class Down(nn.Module):
    def __init__(self, in_channels, out_channels):
        super().__init__()
        self.maxpool_conv = nn.Sequential(
            nn.MaxPool2d(2),
            DoubleConv(in_channels, out_channels)
        )

    def forward(self, x):
        return self.maxpool_conv(x)


class Up(nn.Module):
    def __init__(self, in_channels, out_channels):
        super().__init__()
        self.up = nn.ConvTranspose2d(in_channels, in_channels // 2, kernel_size=2, stride=2)
        self.conv = DoubleConv(in_channels, out_channels)

    def forward(self, x1, x2):
        x1 = self.up(x1)
        diffY = x2.size()[2] - x1.size()[2]
        diffX = x2.size()[3] - x1.size()[3]
        x1 = F.pad(x1, [diffX // 2, diffX - diffX // 2, diffY // 2, diffY - diffY // 2])
        x = torch.cat([x2, x1], dim=1)
        return self.conv(x)


class AttentionBlock(nn.Module):

    def __init__(self, F_g, F_l, F_int):
        super(AttentionBlock, self).__init__()
        self.W_g = nn.Sequential(
            nn.Conv2d(F_g, F_int, kernel_size=1, stride=1, padding=0, bias=True),
            nn.BatchNorm2d(F_int)
        )
        self.W_x = nn.Sequential(
            nn.Conv2d(F_l, F_int, kernel_size=1, stride=1, padding=0, bias=True),
            nn.BatchNorm2d(F_int)
        )
        self.psi = nn.Sequential(
            nn.Conv2d(F_int, 1, kernel_size=1, stride=1, padding=0, bias=True),
            nn.BatchNorm2d(1),
            nn.Sigmoid()
        )
        self.relu = nn.ReLU(inplace=True)

    def forward(self, g, x):
        g1 = self.W_g(g)
        x1 = self.W_x(x)
        psi = self.relu(g1 + x1)
        psi = self.psi(psi)
        return x * psi


class AttentionUNet(nn.Module):

    def __init__(self, n_channels=3, n_classes=1):
        super(AttentionUNet, self).__init__()

        # 编码器
        self.conv1 = DoubleConv(n_channels, 64)
        self.pool1 = nn.MaxPool2d(2)
        self.conv2 = DoubleConv(64, 128)
        self.pool2 = nn.MaxPool2d(2)
        self.conv3 = DoubleConv(128, 256)
        self.pool3 = nn.MaxPool2d(2)
        self.conv4 = DoubleConv(256, 512)
        self.pool4 = nn.MaxPool2d(2)
        self.conv5 = DoubleConv(512, 1024)

        # 注意力模块
        self.att4 = AttentionBlock(F_g=512, F_l=512, F_int=256)
        self.att3 = AttentionBlock(F_g=256, F_l=256, F_int=128)
        self.att2 = AttentionBlock(F_g=128, F_l=128, F_int=64)
        self.att1 = AttentionBlock(F_g=64, F_l=64, F_int=32)

        # 解码器
        self.up4 = Up(1024, 512)
        self.up3 = Up(512, 256)
        self.up2 = Up(256, 128)
        self.up1 = Up(128, 64)

        self.final = nn.Conv2d(64, n_classes, kernel_size=1)

    def forward(self, x):
        # 编码器
        x1 = self.conv1(x)  # 64
        x2 = self.conv2(self.pool1(x1))  # 128
        x3 = self.conv3(self.pool2(x2))  # 256
        x4 = self.conv4(self.pool3(x3))  # 512
        x5 = self.conv5(self.pool4(x4))  # 1024

        # 解码器 + 注意力
        d4 = self.up4(x5, x4)  # 512
        d4 = self.att4(g=d4, x=x4)  # 注意力加权
        d3 = self.up3(d4, x3)  # 256
        d3 = self.att3(g=d3, x=x3)  # 注意力加权
        d2 = self.up2(d3, x2)  # 128
        d2 = self.att2(g=d2, x=x2)  # 注意力加权
        d1 = self.up1(d2, x1)  # 64
        d1 = self.att1(g=d1, x=x1)  # 注意力加权

        out = self.final(d1)
        return out


# ==================== Combo Loss ====================
class ComboLoss(nn.Module):
    def __init__(self, pos_weight=15.0):
        super().__init__()
        self.pos_weight = pos_weight

    def dice_loss(self, pred, target, smooth=1e-5):
        pred = torch.sigmoid(pred)
        pred = pred.contiguous().view(-1)
        target = target.contiguous().view(-1)
        intersection = (pred * target).sum()
        dice = (2. * intersection + smooth) / (pred.sum() + target.sum() + smooth)
        return 1 - dice

    def weighted_bce(self, pred, target):
        weight = torch.where(target == 1, self.pos_weight, 1.0)
        return F.binary_cross_entropy_with_logits(pred, target, weight=weight)

    def forward(self, pred, target):
        return self.dice_loss(pred, target) + self.weighted_bce(pred, target)


# ==================== 评估函数 ====================
def calculate_metrics(pred_logits, true_mask, threshold=0.3):
    pred_prob = torch.sigmoid(pred_logits)
    pred_mask = (pred_prob > threshold).float()
    pred_flat = pred_mask.cpu().numpy().flatten()
    true_flat = true_mask.cpu().numpy().flatten()
    tn, fp, fn, tp = confusion_matrix(true_flat, pred_flat, labels=[0, 1]).ravel()
    precision = tp / (tp + fp) if (tp + fp) > 0 else 0
    recall = tp / (tp + fn) if (tp + fn) > 0 else 0
    f1 = 2 * precision * recall / (precision + recall) if (precision + recall) > 0 else 0
    iou = tp / (tp + fp + fn) if (tp + fp + fn) > 0 else 0
    return {'precision': precision, 'recall': recall, 'f1': f1, 'iou': iou}


# ==================== 训练函数 ====================
def train():
    print("\n加载数据集...")

    sample_mask = cv2.imread(os.path.join(MASKS_FOLDER, os.listdir(MASKS_FOLDER)[0]), cv2.IMREAD_GRAYSCALE)
    void_ratio = np.mean(sample_mask > 0)
    pos_weight = max((1 - void_ratio) / void_ratio, 15.0)
    print(f"空隙像素占比: {void_ratio:.4f}, 正样本权重: {pos_weight:.2f}")

    full_dataset = VoidDataset(IMAGES_FOLDER, MASKS_FOLDER, size=(IMG_SIZE, IMG_SIZE), augment=False)
    all_indices = list(range(len(full_dataset)))
    np.random.seed(42)
    np.random.shuffle(all_indices)
    split_idx = int(len(all_indices) * 0.8)
    train_indices = all_indices[:split_idx]
    val_indices = all_indices[split_idx:]

    train_dataset = VoidDataset(IMAGES_FOLDER, MASKS_FOLDER, size=(IMG_SIZE, IMG_SIZE), augment=True, augment_factor=5)
    val_dataset = VoidDataset(IMAGES_FOLDER, MASKS_FOLDER, size=(IMG_SIZE, IMG_SIZE), augment=False)

    train_augmented_indices = []
    for orig_idx in train_indices:
        for aug_idx in range(5):
            train_augmented_indices.append(orig_idx * 5 + aug_idx)

    train_loader = DataLoader(train_dataset, batch_size=BATCH_SIZE,
                              sampler=torch.utils.data.SubsetRandomSampler(train_augmented_indices),
                              num_workers=4, pin_memory=True)
    val_loader = DataLoader(Subset(val_dataset, val_indices), batch_size=BATCH_SIZE,
                            shuffle=False, num_workers=4, pin_memory=True)

    print(f"原始训练集: {len(train_indices)} 张")
    print(f"增强后训练集: {len(train_augmented_indices)} 张/epoch (5倍增强)")
    print(f"验证集: {len(val_indices)} 张")

    model = AttentionUNet(n_channels=3, n_classes=1).to(DEVICE)
    optimizer = torch.optim.AdamW(model.parameters(), lr=LEARNING_RATE, weight_decay=1e-4)
    scheduler = torch.optim.lr_scheduler.CosineAnnealingLR(optimizer, T_max=EPOCHS)
    criterion = ComboLoss(pos_weight=pos_weight)

    total_params = sum(p.numel() for p in model.parameters())
    print(f"\n模型参数量: {total_params:,} (U-Net: 13,395,329)")
    print(f"\n开始训练 {EPOCHS} 个epoch...")
    print("=" * 60)

    best_f1 = 0
    best_metrics = None
    train_losses = []
    val_losses = []
    f1_scores = []

    start_time = time.time()

    for epoch in range(EPOCHS):
        model.train()
        train_loss = 0
        for images, masks in train_loader:
            images, masks = images.to(DEVICE), masks.to(DEVICE)
            optimizer.zero_grad()
            outputs = model(images)
            loss = criterion(outputs, masks)
            loss.backward()
            optimizer.step()
            train_loss += loss.item()

        avg_train_loss = train_loss / len(train_loader)
        train_losses.append(avg_train_loss)

        model.eval()
        val_loss = 0
        all_outputs = []
        all_targets = []

        with torch.no_grad():
            for images, masks in val_loader:
                images, masks = images.to(DEVICE), masks.to(DEVICE)
                outputs = model(images)
                loss = criterion(outputs, masks)
                val_loss += loss.item()
                all_outputs.append(outputs.cpu())
                all_targets.append(masks)

        avg_val_loss = val_loss / len(val_loader)
        val_losses.append(avg_val_loss)

        all_outputs = torch.cat(all_outputs, dim=0)
        all_targets = torch.cat(all_targets, dim=0)
        metrics = calculate_metrics(all_outputs, all_targets, threshold=CONFIDENCE_THRESHOLD)
        f1_scores.append(metrics['f1'])

        if metrics['f1'] > best_f1:
            best_f1 = metrics['f1']
            best_metrics = metrics
            torch.save(model.state_dict(), os.path.join(OUTPUT_FOLDER, 'attention_unet_best.pth'))

        scheduler.step()

        if (epoch + 1) % 10 == 0:
            print(
                f"Epoch {epoch + 1:3d}/{EPOCHS} | Train Loss: {avg_train_loss:.4f} | Val Loss: {avg_val_loss:.4f} | F1: {metrics['f1']:.4f} | Prec: {metrics['precision']:.4f} | Rec: {metrics['recall']:.4f}")

    training_time = time.time() - start_time

    print("\n" + "=" * 60)
    print(f"✅ Attention U-Net 训练完成！")
    print(f"  最佳F1: {best_f1:.4f}")
    print(f"  对应精确率: {best_metrics['precision']:.4f}")
    print(f"  对应召回率: {best_metrics['recall']:.4f}")
    print(f"  对应IoU: {best_metrics['iou']:.4f}")
    print(f"  训练时间: {training_time / 60:.2f}分钟")
    print(f"  结果保存至: {OUTPUT_FOLDER}")
    print("=" * 60)

    plt.figure(figsize=(12, 5))
    plt.subplot(1, 2, 1)
    plt.plot(train_losses, label='Train Loss')
    plt.plot(val_losses, label='Val Loss')
    plt.xlabel('Epoch')
    plt.ylabel('Loss')
    plt.title('Attention U-Net Loss Curves')
    plt.legend()
    plt.grid(True)

    plt.subplot(1, 2, 2)
    plt.plot(f1_scores, label='F1 Score')
    plt.xlabel('Epoch')
    plt.ylabel('F1')
    plt.title(f'Attention U-Net F1 Curve (Best: {best_f1:.4f})')
    plt.legend()
    plt.grid(True)

    plt.savefig(os.path.join(OUTPUT_FOLDER, 'attention_unet_curves.png'), dpi=150)
    plt.close()

    with open(os.path.join(OUTPUT_FOLDER, 'attention_unet_results.txt'), 'w') as f:
        f.write("Attention U-Net 训练结果\n")
        f.write("=" * 50 + "\n")
        f.write(f"最佳F1: {best_f1:.4f}\n")
        f.write(f"精确率: {best_metrics['precision']:.4f}\n")
        f.write(f"召回率: {best_metrics['recall']:.4f}\n")
        f.write(f"IoU: {best_metrics['iou']:.4f}\n")
        f.write(f"训练时间: {training_time / 60:.2f}分钟\n")


if __name__ == "__main__":
    train()
